251 void GraspVisualization::addSuperq(const Vector &x, ImageOf<PixelRgb> &imgOut,const int &col)
252 {
253  int count=0;
254 
255  PixelRgb color(col,0,0);
256 
257  if (col==0)
258  {
259  color.r=0;
260  color.b=255;
261  }
262 
263  Vector pos, orient;
264  double co,so,ce,se;
265  Stamp *stamp=NULL;
266 
267  Matrix R=euler2dcm(x.subVector(8,10));
268  R=R.transposed();
269 
270  if ((norm(x)>0.0))
271  {
272  if (eye=="left")
273  {
274  if (igaze->getLeftEyePose(pos,orient,stamp))
275  {
276  H=axis2dcm(orient);
277  H.setSubcol(pos,0,3);
278  H=SE3inv(H);
279  }
280  }
281  else
282  {
283  if (igaze->getRightEyePose(pos,orient,stamp))
284  {
285  H=axis2dcm(orient);
286  H.setSubcol(pos,0,3);
287  H=SE3inv(H);
288  }
289  }
290 
291  Vector point(3,0.0);
292  Vector point2D(2,0.0);
293  double step=2*M_PI/50;
294 
295  for (double eta=-M_PI; eta<M_PI; eta+=step)
296  {
297  for (double omega=-M_PI; omega<M_PI;omega+=step)
298  {
299  co=cos(omega); so=sin(omega);
300  ce=cos(eta); se=sin(eta);
301 
302  point[0]=x[0] * sign(ce)*(pow(abs(ce),x[3])) * sign(co)*(pow(abs(co),x[4])) * R(0,0) +
303  x[1] * sign(ce)*(pow(abs(ce),x[3]))* sign(so)*(pow(abs(so),x[4])) * R(0,1)+
304  x[2] * sign(se)*(pow(abs(se),x[3])) * R(0,2) + x[5];
305 
306  point[1]=x[0] * sign(ce)*(pow(abs(ce),x[3])) * sign(co)*(pow(abs(co),x[4])) * R(1,0) +
307  x[1] * sign(ce)*(pow(abs(ce),x[3])) * sign(so)*(pow(abs(so),x[4])) * R(1,1)+
308  x[2] * sign(se)*(pow(abs(se),x[3])) * R(1,2) + x[6];
309 
310  point[2]=x[0] * sign(ce)*(pow(abs(ce),x[3])) * sign(co)*(pow(abs(co),x[4])) * R(2,0) +
311  x[1] * sign(ce)*(pow(abs(ce),x[3])) * sign(so)*(pow(abs(so),x[4])) * R(2,1)+
312  x[2] * sign(se)*(pow(abs(se),x[3])) * R(2,2) + x[7];
313 
314  point2D=from3Dto2D(point);
315 
316  cv::Point target_point((int)point2D[0],(int)point2D[1]);
317 
318  if ((target_point.x<0) || (target_point.y<0) || (target_point.x>=320) || (target_point.y>=240))
319  {
320  count++;
321  }
322  else
323  imgOut.pixel(target_point.x, target_point.y)=color;
324  }
325  }
326  }
327 }

330 Vector GraspVisualization::from3Dto2D(const Vector &point3D)
331 {
332  Vector point2D(3,0.0);
333  Vector point_aux(4,1.0);
334  point_aux.setSubvector(0,point3D);
335  point2D=K*H*point_aux;
336  return point2D.subVector(0,1)/point2D[2];
337 }
