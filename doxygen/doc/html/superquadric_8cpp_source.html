45 void grasping_NLP::init(const Vector &objectext, Vector &handext, int &n_handpoints, const string &str_hand)
46 {

56  euler[1]=object[9];

59  euler[0]=object[5];
60  euler[1]=object[6];
61  euler[2]=object[7];
62  H_o2w.setSubcol(euler,0,3);
63 
64  euler[0]=hand[8];
65  euler[1]=hand[9];
66  euler[2]=hand[10];
67  H_h2w=euler2dcm(euler);
68  euler[0]=hand[5];
69  euler[1]=hand[6];
70  euler[2]=hand[7];
71  H_h2w.setSubcol(euler,0,3);
72 
73  for(int i=0; i<(int)sqrt(n_handpoints); i++)
74  {
75  for (double theta=-M_PI/2; theta<=0; theta+=M_PI/((int)sqrt(n_handpoints)))

85 Vector grasping_NLP::computePointsHand(Vector &hand, int j, int l, const string &str_hand, double &theta)
86 {
87  Vector point(3,0.0);
88  double omega;
89  double ce,se,co,so;
90 
91  if (findMax(object.subVector(0,2))> findMax(hand.subVector(0,2)))
92  hand[1]=findMax(object.subVector(0,2));
93 
94  if (str_hand=="right")
95  {
96  omega=j*2*M_PI/(l);
97 
98  ce=cos(theta);
99  se=sin(theta);
100  co=cos(omega);
101  so=sin(omega);
102 
103  point[0]=hand[0] * sign(ce)*(pow(abs(ce),hand[3])) * sign(co)*(pow(abs(co),hand[4]));
104  point[1]=hand[1] * sign(ce)*(pow(abs(ce),hand[3])) * sign(so)*(pow(abs(so),hand[4]));
105  point[2]=hand[2] * sign(se)*(pow(abs(se),hand[3]));
106  }
107  else
108  {
109  omega=j*2*M_PI/(l);
110 
111  ce=cos(theta+M_PI/2);
112  se=sin(theta+M_PI/2);
113  co=cos(omega);
114  so=sin(omega);
115 
116  point[0]=hand[0] * sign(ce)*(pow(abs(ce),hand[3])) * sign(co)*(pow(abs(co),hand[4]));
117  point[1]=hand[1] * sign(ce)*(pow(abs(ce),hand[3])) * sign(so)*(pow(abs(so),hand[4]));
118  point[2]=hand[2] * sign(se)*(pow(abs(se),hand[3]));
119  }
120 
121  Vector point_tr(4,0.0);
122 
123  euler[0]=hand[8];
124  euler[1]=hand[9];
125  euler[2]=hand[10];
126  H_h2w=euler2dcm(euler);
127  euler[0]=hand[5];
128  euler[1]=hand[6];
129  euler[2]=hand[7];
130  H_h2w.setSubcol(euler,0,3);
131 
132  Vector point_tmp(4,1.0);
133  point_tmp.setSubvector(0,point);
134  point_tr=H_h2w*point_tmp;
135  point=point_tr.subVector(0,2);
136 
137  return point;
138 }

211  /****************************************************************/
212  double grasping_NLP::f(Vector &obj, const Ipopt::Number *x, Vector &point)
213  {
214  Matrix H(4,4);
215 
216  Vector point_tr(4,0.0);
217  Vector point_tmp(4,1.0);
218  point_tmp.setSubvector(0,point);
219 
220  euler[0]=x[3];
221  euler[1]=x[4];
222  euler[2]=x[5];
223  H_x=euler2dcm(euler);
224  euler[0]=x[0];
225  euler[1]=x[1];
226  euler[2]=x[2];
227  H_x.setSubcol(euler,0,3);
228 
229  point_tr=H_x*point_tmp;
230 
231  double num1=H_o2w(0,0)*point_tr[0]+H_o2w(0,1)*point_tr[1]+H_o2w(0,2)*point_tr[2]-obj[5]*H_o2w(0,0)-obj[6]*H_o2w(0,1)-obj[7]*H_o2w(0,2);
232  double num2=H_o2w(1,0)*point_tr[0]+H_o2w(1,1)*point_tr[1]+H_o2w(1,2)*point_tr[2]-obj[5]*H_o2w(1,0)-obj[6]*H_o2w(1,1)-obj[7]*H_o2w(1,2);
233  double num3=H_o2w(2,0)*point_tr[0]+H_o2w(2,1)*point_tr[1]+H_o2w(2,2)*point_tr[2]-obj[5]*H_o2w(2,0)-obj[6]*H_o2w(2,1)-obj[7]*H_o2w(2,2);
234 
235  double tmp=pow(abs(num1/obj[0]),2.0/obj[4]) + pow(abs(num2/obj[1]),2.0/obj[4]);
236 
237  return pow( abs(tmp),obj[4]/obj[3]) + pow( abs(num3/obj[2]),(2.0/obj[3]));
238  }

292  bool grasping_NLP::eval_grad_f(Ipopt::Index n, const Ipopt::Number* x, bool new_x,
293  Ipopt::Number *grad_f)
294  {
295  Vector x_tmp(6,0.0);
296  double grad_p, grad_n;
297  double eps=1e-6;
298 
299  for(Ipopt::Index i=0;i<n;i++)
300  x_tmp[i]=x[i];
301 
302  for(Ipopt::Index j=0;j<n;j++)
303  {
304  x_tmp[j]=x_tmp[j]+eps;
305 
306  grad_p=F_v(x_tmp,points_on);
307 
308  x_tmp[j]=x_tmp[j]-eps;
309 
310  grad_n=F_v(x_tmp,points_on);
311 
312  grad_f[j]=(grad_p-grad_n)/eps;
313  }
314 
315  return true;
316  }

319  bool grasping_NLP::eval_g(Ipopt::Index n, const Ipopt::Number *x, bool new_x,
320  Ipopt::Index m, Ipopt::Number *g)
321  {
322  euler[0]=x[3];
323  euler[1]=x[4];
324  euler[2]=x[5];
325  H_x=euler2dcm(euler);
326  euler[0]=x[0];
327  euler[1]=x[1];
328  euler[2]=x[2];
329  H_x.setSubcol(euler,0,3);
330 
331  Matrix H(4,4);
332  H=H_x*H_h2w;
333 
334  g[0]=H(2,2);
335  g[1]=H(0,0);
336  g[2]=H(1,2);
337  g[3]=H(1,0);
338 
339  Vector x_min;
340  double minz=10.0;
341 
342  for (size_t i=0; i<points_on.size(); i++)
343  {
344  Vector pnt(4,1.0);
345  pnt.setSubvector(0,points_on[i]);
346  Vector point=H_x*pnt;
347 
348  if (point[2]<minz)
349  {
350  minz=point[2];
351  x_min=point;
352  }
353  }
354 
355  g[4]=plane(0,0)*x_min[0]+plane(1,0)*x_min[1]+plane(2,0)*x_min[2]+plane(3,0);
356 
357  Vector robotPose(3,0.0);
358  Vector x_tmp(6,0.0);
359  x_tmp[0]=x[0];
360  x_tmp[1]=x[1];
361  x_tmp[2]=x[2];
362  x_tmp[3]=x[3];
363  x_tmp[4]=x[4];
364  x_tmp[5]=x[5];
365 
366  if (l_o_r=="right")
367  robotPose=x_tmp.subVector(0,2)-hand[0]*(H.getCol(2).subVector(0,2));
368  else
369  robotPose=x_tmp.subVector(0,2)+hand[0]*(H.getCol(2).subVector(0,2));
370 
371  g[5]=object[0]*object[1]*object[2]*(pow(f_v2(object,x_tmp, robotPose), object[3]) -1);
372 
373  return true;
374  }
